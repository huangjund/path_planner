295  struct MotionCompare
296  {
298  MotionCompare(base::OptimizationObjectivePtr opt, base::ProblemDefinitionPtr pdef)
299  : opt_(std::move(opt)), pdef_(std::move(pdef))
300  {
301  }
302 
304  inline base::Cost costPlusHeuristic(const Motion *m) const
305  {
306  return opt_->combineCosts(m->cost, opt_->costToGo(m->state, pdef_->getGoal().get()));
307  }
308 
310  inline base::Cost alphaCostPlusHeuristic(const Motion *m, double alpha) const
311  {
312  return opt_->combineCosts(base::Cost(alpha * m->cost.value()),
313  opt_->costToGo(m->state, pdef_->getGoal().get()));
314  }
315 
317  inline bool operator()(const Motion *m1, const Motion *m2) const
318  {
319  // we use a max heap, to do a min heap so the operator < returns > in order to make it a min heap
320  return !opt_->isCostBetterThan(costPlusHeuristic(m1), costPlusHeuristic(m2));
321  }
